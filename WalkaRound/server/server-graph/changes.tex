\subsection{graph}
\paragraph{Vertex}

 \item Das Attribut ID ist nun vom Typ int.
\end{itemize}

\paragraph{Edge}
\begin{itemize}
 \item Das Attribut ID ist nun vom Typ int.
 \item Der Rückgabewert von getLength() ist nun vom Typ double.
\end{itemize}

\paragraph{Graph}
\begin{itemize}
 \item Der Konstruktur erhält eine Liste von Edge-Objekten.
\end{itemize}
